154  int exact_match) {
155  char* configname;
156  if (prefix != NULL)
157  asprintf(&configname, "%s.%s", prefix, source);
158  else

179 // this is the callback for inih which is called for each configuration line:
180 // - first fetch the relevant configuration by exact match
181 // - if no config was found already, create one with default values
182 // - the parse configuration value
183 static int _unclog_config_handler(void* user, const char* section, const char* name,
184  const char* value) {
185  (void)user;
186  if (LOG()) fprintf(stderr, "unclog.config: %s: %s=%s\n", section, name, value);
187 
188  const char* configname = "";
189  if (!MATCH(section, "defaults")) {
190  configname = section;
191  }
192  unclog_config_t* c = _unclog_config_get(NULL, configname, 1);
193  if (c == NULL) {
194  unclog_config_t* defaults = _unclog_config_get(NULL, "", 1);
195  c = _unclog_config_create(configname, defaults->level, defaults->details);
196  LIST_INSERT_HEAD(&unclog_global->configs, c, entries);
197  }
198 
199  if (MATCH(section, "defaults") && MATCH(name, "Internal")) {
200  if (MATCH(value, "Log")) unclog_global->flags = UNCLOG_FLAGS_LOG;
201  } else if (MATCH(name, "Level")) {
202  int level = unclog_level(value);
203  if (level != -1) c->level = level;
204  } else if (MATCH(name, "Details")) {

206  char* v = strdup(value);

209  while (tok != NULL) {

213  free(v);
214  } else {

216  v->next = c->config;
217  c->config = v;
218  }

220  return 0;
221 }

294 // update the configuration store, this does not touch the sources or sinks,
295 // just the config
296 // - first clear up the config and install default config (this default might
297 // be updated by the config but at least the default must be there)
298 // - if the input config pointer is not NULL, us that config
299 // - else search for files in unclog_config_locations
300 static void _unclog_config(const char* config) {
301  _unclog_config_clear(0);
302 
303  unclog_config_t* c = _unclog_config_create("", UNCLOG_LEVEL_DEFAULT, UNCLOG_DETAILS_DEFAULT);
304  LIST_INSERT_HEAD(&unclog_global->configs, c, entries);
305 
306  if (config != NULL) {
307  ini_parse_string(config, _unclog_config_handler, NULL);
308  } else {
309  const char** f = unclog_config_locations;
310  for (; *f != NULL; f++) {
311  if (access(*f, R_OK) == 0) {
312  if (LOG()) fprintf(stderr, "unclog: read config from %s\n", *f);
313  ini_parse(*f, _unclog_config_handler, NULL);
314  break;
315  }
316  }
317  }
318 
319  c = unclog_global->configs.lh_first;
320  for (; c != NULL; c = c->entries.le_next) {
321  if (strncmp(c->name, "sink.", 5) != 0) continue;
322  unclog_sink_t* sink = _unclog_sink_get(c->name);
323  if (sink != NULL) continue;
324  char* type = unclog_config_value_get(c->config, "Type");
325  if (type == NULL) continue;
326  for (unclog_sink_list_t* l = unclog_sink_list; l->name != NULL; l++) {
327  if (!MATCH(type, l->name)) continue;

329  }

384 // the actual logging call:
385 // - check if handle is valid, and if the level checks out
386 // - fill up data structure
387 // - call all the logging sinks
388 void unclog_log(unclog_data_t data, ...) {
389  if (data.ha == NULL) return;
390  unclog_source_t* source = (unclog_source_t*)data.ha;
391 
392  if (!UNCLOG_LEVEL_COMPARE(data.le, source->level)) return;
393 
394  time_t now = time(NULL);
395  gmtime_r(&now, &data.ti);
396  data.so = source->name;
397 
398  pthread_rwlock_rdlock(&unclog_mutex);
399  unclog_sink_t* sink = unclog_global->sinks.lh_first;
400  for (; sink != NULL; sink = sink->entries.le_next) {
401  if (!UNCLOG_LEVEL_COMPARE(data.le, sink->level)) continue;
402 
403  data.si = sink->internal;
404 
405  va_list list;
406  va_start(list, data);
407  sink->methods.log(&data, list);
408  va_end(list);
409  }
410  pthread_rwlock_unlock(&unclog_mutex);
411 }
